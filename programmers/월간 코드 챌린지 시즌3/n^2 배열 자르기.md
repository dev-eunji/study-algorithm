fun solution(n: Int, left: Long, right: Long): IntArray {
    val answer = arrayListOf<Int>()
    for(i in left..right) {
        val row = (i / n).toLong()
        val col = (i % n).toLong()
        answer.add((Math.max(row, col) + 1).toInt())
    }
    return answer.toIntArray()
}
