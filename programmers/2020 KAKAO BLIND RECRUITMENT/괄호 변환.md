private fun check(s: String): Boolean {
    val stack = Stack<Char>()
    s.forEach { c ->
        if (c == '(') {
            stack.push(c)
        } else {
            if (stack.empty()) {
                return false
            } else {
                stack.pop()
            }
        }
    }
    return true
}

private fun divide(s: String): Int {

    }
