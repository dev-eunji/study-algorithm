fun solution(n: Int, times: IntArray): Long {
    var answer = 0L
    var min = 0                                     // ìµœì†Œ ì‹œê°„

        if (count < n) {
            min = mid + 1
        } else {
            max = mid - 1
            answer = mid
        }
    }
    return answer
}
