    val answer = IntArray(2) {0}
    fun dfs(row: Int, col: Int, size: Int, arr: Array<IntArray>) {

        var check = true
        val value = arr[row][col]
        for(i in row until row+size) {
            if(!check) break
            for(j in col until col+size) {
                if(arr[i][j] != value) {
                    check = false
                    break
                }
            }
        }
        if(check) {
            if (value == 0) answer[0]++
            else answer[1]++
            return
